<script>

// Sentences based on finding

const findingsTexts = {

  Liver: {

    Normal: "The liver is normal in size and appearance, with no focal lesions.",

    Lesion: "There is a hypodense lesion seen in the liver, suggestive of a cyst.",

    Enlarged: "The liver is enlarged (hepatomegaly) without focal lesions."

  },

  Gallbladder: {

    Normal: "The gallbladder is unremarkable, with no stones or wall thickening.",

    Gallstones: "Multiple gallstones are present within the gallbladder.",

    Inflamed: "The gallbladder wall is thickened, suggestive of cholecystitis."

  },

  Pancreas: {

    Normal: "The pancreas appears normal in size and contour, with no masses.",

    Mass: "A solid mass is noted in the pancreatic head region.",

    Atrophic: "The pancreas is atrophic, consistent with chronic pancreatitis."

  }

};


function generateReport() {

  const organs = document.querySelectorAll('.organ');

  let report = '';


  organs.forEach(select => {

    const organ = select.dataset.organ;

    const finding = select.value;

    const sentence = findingsTexts[organ][finding];

    report += sentence + '\n';

  });


  document.getElementById('report').value = report;

}
